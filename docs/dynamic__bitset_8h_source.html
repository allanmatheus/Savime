248  template <typename BlockInputIterator>
249  void m_append(BlockInputIterator first, BlockInputIterator last, std::forward_iterator_tag)
250  {
251  assert(first != last);
252  block_width_type r = count_extra_bits();
253  std::size_t d = boost::detail::distance(first, last);
254  m_bits.reserve(num_blocks() + d);
255  if (r == 0) {
256  for( ; first != last; ++first)
257  m_bits.push_back(*first); // could use vector<>::insert()
258  }
259  else {
260  m_highest_block() |= (*first << r);
261  do {
262  Block b = *first >> (bits_per_block - r);
263  ++first;
264  m_bits.push_back(b | (first==last? 0 : *first << r));
265  } while (first != last);
266  }
267  m_num_bits += bits_per_block * d;
268  }

801 template <typename Block, typename Allocator>
802 void dynamic_bitset<Block, Allocator>::
803 resize(size_type num_bits, bool value) // strong guarantee
804 {
805 
806  const size_type old_num_blocks = num_blocks();
807  const size_type required_blocks = calc_num_blocks(num_bits);
808 
809  const block_type v = value? ~Block(0) : Block(0);
810 
811  if (required_blocks != old_num_blocks) {
812  m_bits.resize(required_blocks, v); // s.g. (copy)
813  }
814 
815 
816  // At this point:
817  //
818  // - if the buffer was shrunk, we have nothing more to do,
819  // except a call to m_zero_unused_bits()
820  //
821  // - if it was enlarged, all the (used) bits in the new blocks have
822  // the correct value, but we have not yet touched those bits, if
823  // any, that were 'unused bits' before enlarging: if value == true,
824  // they must be set.
825 
826  if (value && (num_bits > m_num_bits)) {
827 
828  const block_width_type extra_bits = count_extra_bits();
829  if (extra_bits) {
830  assert(old_num_blocks >= 1 && old_num_blocks <= m_bits.size());
831 
832  // Set them.
833  m_bits[old_num_blocks - 1] |= (v << extra_bits);
834  }
835 
836  }
837 
838  m_num_bits = num_bits;
839  m_zero_unused_bits();
840 
841 }

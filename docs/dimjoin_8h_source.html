62  || IN_RANGE(realLower[1], realLower[0], realUpper[0]) )

65  joinedRange->lower_bound = (realLower[0] > realLower[1]) ? realLower[0] : realLower[1];
66  joinedRange->upper_bound = (realUpper[0] < realUpper[1]) ? realUpper[0] : realUpper[1];
67  intersection[dim->name] = joinedRange;

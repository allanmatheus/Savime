83  void Configure()
84  {

86  for(int32_t i = 0; i < numDims; i++)
87  {
88  auto dimName = _dimensions[i]->name;
89 
90  _multipliers[dimName] = 1;
91  _skew[dimName] = _dimensions[i]->GetLength();
92  _adj[dimName] = 1;
93 
94  for(int32_t j = i+1; j < numDims; j++)
95  {
96  int64_t preamble = _multipliers[dimName];
97  _multipliers[dimName] = preamble*_dimensions[j]->GetLength();
98  _skew[dimName] *= _dimensions[j]->GetLength();
99  _adj[dimName] *= _dimensions[j]->GetLength();
100  }
101  }
102 
103  if(numDims == 0)
104  {
105  _skew["i"] = 1.0;
106  _adj["i"] = 1.0;
107  }
108  }
